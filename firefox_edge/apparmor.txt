#include <tunables/global>

# Hardened AppArmor profile for the Firefox (Edge) Home Assistant add-on.
#
# The profile name MUST match the add-on slug ("firefox_edge"), otherwise the
# Supervisor cannot load it.
#
# Like the stable Firefox add-on, this variant runs the browser as an
# unprivileged user. It additionally runs `apk update && apk upgrade` at init
# (as root, via a cont-init.d script) to pull the latest Firefox from Alpine
# edge; that package operation relies on the broad file access below.
#
# NOTE: this cannot be validated in CI (the build host has no AppArmor LSM). If
# the add-on fails to start, inspect the host log for the denial
# (`dmesg | grep -i 'apparmor="DENIED".*profile="firefox_edge"'`) and widen the
# offending rule. Last resort: add `apparmor: false` to config.yaml.
profile firefox_edge flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  # Firefox and the apk upgrade need full network access.
  network,

  # Capabilities — Docker's default set (present in every add-on container)
  # plus sys_ptrace for the crash reporter. This does NOT grant sys_admin,
  # sys_module, sys_rawio, etc.
  capability setuid,        # init drops privileges to the app user
  capability setgid,
  capability setpcap,       # init drops capabilities
  capability setfcap,
  capability chown,         # apk/init take ownership of files
  capability fowner,
  capability fsetid,
  capability dac_override,
  capability kill,          # supervisor stops services
  capability mknod,         # device nodes
  capability sys_chroot,    # Firefox content sandbox
  capability net_bind_service,
  capability net_raw,
  capability audit_write,
  capability sys_ptrace,    # Firefox crash reporter

  # Firefox's content sandbox sets up private mounts inside a namespace.
  mount,
  umount,
  pivot_root,
  ptrace (read, trace) peer=firefox_edge,
  signal (send, receive) peer=firefox_edge,
  signal (send) set=(kill,term,int,hup,cont,usr1,usr2),

  # Broad file access (read/write/exec/mmap). Required in particular so the
  # init-time `apk upgrade` can rewrite the system directories. The denials
  # below carve out the clearly out-of-scope, sensitive locations.
  file,

  # Hardening denials — take precedence over the broad `file` rule above.
  deny /root/** rwklx,
  deny @{PROC}/sysrq-trigger rwklx,
  deny /sys/kernel/security/** rwklx,
}
